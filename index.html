Click to select the boxes!

Click and hold left mouse to make a multi-selection!

Hold ctrl to make multiple selections.

Comment tutorial to realize the demo 
const options = {

    // Class for the selection-area
    class: 'selection',

    // All elements in this container can be selected
    containers: ['.box-wrap'],

    // The container is also the boundary in this case
    boundarys: ['.box-wrap'],
    // Or
    boundaries: ['.box-wrap'],

    onSelect(evt) {

         // Get elements which has been selected so far
         const selectedElements = evt.selectedElements;

    },

    onStart(evt) {

        // Get elements which has been selected so far
        const selectedElements = evt.selectedElements;

    },

    onMove(evt) {

        // Get the currently selected elements and those
        // which where removed since the last selection.
        const selectedElements = evt.selectedElements;
        const removedElements = evt.changedElements.removed;

    },

    onStop(evt) {

         // Get elements which has been selected so far
         const selectedElements = evt.selectedElements;

    },
};

Selection.create(options);
